\hypertarget{gpio_pin_layout_page_autotoc_md14}{}\doxysubsubsection{\texorpdfstring{Notes}{Notes}}\label{gpio_pin_layout_page_autotoc_md14}

\begin{DoxyItemize}
\item Pins marked as "{}\+Free"{} are currently unused.
\item Strapping pins are not used.
\item All inputs use internal pull-\/ups. -\/-\/\texorpdfstring{$>$}{>} active-\/low
\item All extern outputs are active-\/high and drive low-\/side loads via IRLZ44N MOSFETs.
\item All extern inputs are active-\/low and protected via PC817 Optocoupler. 
\end{DoxyItemize}
